  142. myTable = {} --[[
  143.   This is the only default data structure in Lua. This can be associated as an array or a table.
  144.  
  145.   When treated as a table, we use mostly non-numeric String based keys.
  146.  
  147.   When treated as an array, we use numeric based keys.
  148.  
  149.   [!] VERY IMPORTANT. REMEMBER THAT LUA TREATS THE FIRST INDEX OF A TABLE/ARRAY AS "1" NOT "0"
  150.  
  151.   There are no generic array types in Lua and thus we can mix both numeric and String based
  152.   key values in Lua.
  153.  
  154.   They are declared with "{}"
  155.  
  156.   We can access and manipulate content within the structure using keys. Done so by following
  157.   the format:
  158.  
  159. 1| table[key] = value
  160.  
  161. ^^^^^
  162. Notice the usage of the "[]" square brackets
  163. ]]
  164. myTable["myIndex"] = 100 -- Key "myIndex" set to "100" in table "myTable"
  165. myTable[1] = 20       -- Key "1" set to "20" in table "myTable"
  166.  
  167. --[[
  168.   To access data, we use the same format, except now we don't use the assignment operator "="
  169.  
  170.   If no data is present with that key, Lua will return "nil".
  171.  
  172.   We can also delete values by just setting them to "nil"
  173. ]]
  174. print(myTable[1]) -- 20 -> Based on what we set above
  175. print(myTable["thisKeyExists?"]) -- nil -> We did not set anything for this key above
  176.  
  177. myTable.myIndex = 30 --[[
  178.   This is the other method to select values, this is the same as:
  179.  
  180. 1| myTable["myIndex"]
  181.  
  182.   ^^^^
  183.   Note this is not the same as myTable[myIndex]
  184. ]]
  185.  
  186. myTable2 = {first = 1, second = 2} --[[
  187.   Initialization can be done so this way, which would be the same as:
  188.  
  189.    1| myTable2 = {["first"] = 1, ["second"] = 2}
  190.  
  191. ]]
  192.  
  193. myArrayTable = {53, 34, 12} --[[
  194.   This can also be declared as
  195.  
  196.    1| myArrayTable = {[1] = 53, [2] = 34, [3] = 12}
  197.  
  198. ]]
  199. print(myArrayTable[1]) -- Would print "53" because the first element is 53

  201. print(#myArrayTable) --[[
  202.   This gives us the length of all numerical keys (integer keys) in the array.

  204.   The "#" operator for this operation:
  205.  
  206. 1| print(#{["hi"] = 1, 32, 4}) -- 2 Because "hi" is not numeric and will not be counted

  208. ]]

  211. --[[ SECTION ARITHMETICS & Operations]]--
  212. -----------------------------------------
  213. --[[
  214.   Basic Arithmetics can be used on Number types and between Number types but never between
  215.   any other.
  216.  
  217.   Operators can be used between variables or raw values.
  218.  
  219.   Operators are:
  220.  
  221.   + -> For addition                    | 2+3 => 5
  222.   - -> For subtraction                 | 2-3 => -1
  223.   * -> For multiplication              | 2*3 => 6
  224.   / -> For division                    | 2/3 ~> 0.67
  225.   ^ -> For exponent (power) [NOT BIT XOR] | 2^3 => 8
  226.   % -> For remainder (modulo)          | 2%3 => 2
  227.   - -> For value Negation              | -(2+3) => -5
  228.  
  229.   THERE ARE NO POSTFIX OPERATORS IN LUA (i++, c--, ++a, ++b), thus you have to do (for maybe incrementation by 1):
  230. my_Var = my_Var + 1
  231.  
  232.   Operator Precedence (highest [top] to lowest [bottom])
  233.   "^"
  234.   "not"
  235.   "*","/"
  236.   "+","-"
  237.   "<",">","<=","=>","==","~="
  238.   "and"
  239.   "or"
  240.  
  241.   [!] Same operator precedence at the same level is executed based on their placement from left to right
  242.  
  243. ]]
  244.  
  245. print("Hello World" .. "Its me") --[[
  246. String concat, or adding two strings together is accomplished using the ".." operator
  247.  
  248. Using this operation you can also "coerce" other types to a "String", take for example:
  249.  
  250.      1| var_Num = 3
  251.      2| print("My Number: " .. varNum) -- Effectively coerce "var_Num" to be a String via String concat
  252.  
  253. ]]
  254.  
  255. print("Hello World" + "its me") --[[
  256.   String concat can also be achieved via the "+" addition operator and also has the same effects
  257.   as ".."
  258.  
  259.   However, remember that when both sides are numbers, the standard addition operation is performed.
  260.  
  261.   Note:
  262.    print(1 .. 3) -> 13
  263.    print(1 + 3)  -> 4
  264. ]]

  268. --[[ SECTION SCOPE ]]--
  269. -----------------------
  270. --[[
  271.   Scope can be defined as something's prevalence or visibility somewhere.
  272.  
  273. Hypothetical code:
  274.  
  275.      1| var b
  276.      2| block
  277.      3|   var a
  278.      4| end
  279.  
  280. Variable "b" is alive from lines 1-4 while variable "a" is only alive within 2-3 (exclusive)
  281.  
  282. In Lua, a variable without a scope modifier is referred to as a global variable.
  283.  
  284. To avoid conflicts between naming we add the "local" scope modifier in front of a variable name:
  285.  
  286.    1| local myVariable = nil
  287.  
  288. ^^^^
  289. Local variables only exist in their permitted block and nowhere else.
  290.  
  291. [!] Every loop iteration and function call creates a new scope with new variables.
  292. [!] Generally you want to use "local" specifiers as much as possible in order to:
  293.          1. Make your code more readable
  294.          2. Create less trap variables that can lead to undesired behaviors.
  295. ]]
  296.  
  297. local t = 3
  298. t = t + 2 -- Here t is referenced and a global reference is not created.
  299.  
  300. local function f() end
  301. local f2
  302. f2 = function() end
  303. local f3 = function() end --[[
  304.   This section from lua-users.org:
  305.   """
  306.   The difference between the last two examples is important: the local variable still doesn't exist to the right of the = that gives it the initial value. So if the contents of the function used f to get a reference to itself, it will correctly get the local variable in the first and second versions, but the third version will get the global f (which will be nil, if not a completely unrelated value set by some other code).
  307.   """
  308. ]]
  309.  
  310. --[[
  311.   Closures are the effect that functions (local or not) can use locale variables outside; these functions
  312.   are therefore referred to as "closures"
  313.    -The function sees any changes to the variable outside
  314. ]]

  318. --[[SECTION CONTROLS & CONDITIONS ]]--
  319. --------------------------------------
  320. --[[
  321.   For any boolean expression Lua treats:
  322.  
  323. nil, false -> FALSE
  324. Anything else -> TRUE
  325.  
  326. Statements like variable assignment are not treated as valid boolean expressions
  327. ]]
  328. myBool = true
  329. myVar = 3
  330.  
  331. if myBool then -- Note for boolean expressions, we don't have to do "myBool == true", you can, but is less concise
  332.   print("I AM HUNGRY")
  333. end --[[
  334.   The most basic conditional statement is the "if" which follows the following format:
  335.  
  336.    1| if boolean-condition then -- If condition declaration
  337.    2|   todo                 -- Statement to run on boolean condition for true
  338.    3| end                    -- Marks the end of this case
  339.  
  340. -If "boolean-condition" is true, then the "todo" code runs
  341.   In the above code snippet, since "myBool" is true, then the program will print "I AM HUNGRY"
  342. ]]
  343.  
  344. if myBool then
  345.   print("I am hungry")
  346. else
  347.   print("I am not hungry")
  348. end --[[
  349.   When we want to introduce a "default" case, we add the else keyword for when the other conditions fail.
  350.  
  351. There can only be one else statement per conditional statement
  352.  
  353.   [!] Note that the "else" statement does not utilize a "then" keyword
  354. ]]
  355.  
  356. if myVar == 2 then
  357.   print("2")
  358. elseif myVar >= 3 then
  359.   print(">=3")
  360. else
  361.   print("?")
  362. end --[[
  363.   We can add intermediate conditional cases using the "elseif [expr] then" expression
  364.   which is just an "if" statement but renamed.
  365.  
  366.   Processing this conditional block goes from top to bottom.
  367.  
  368.   We could've also ended the entire conditional block at the "else-if" with an "end" and not include a
  369.   default "else" case
  370. ]]
  371.  
  372. if myVar == 2 then
  373.   print("My var is 2!!")
  374. end --[[
  375.   Conditional operators are used to validate certain things. They are parsed as "boolean expressions" that
  376.   express "booleans".
  377.  
  378.   Here are all of the conditional operators in Lua:
  379. X > Y  -> X greater than Y
  380. X >= Y -> X greater than or equal to Y
  381. X < Y  -> X smaller than Y
  382. X <= Y -> X smaller than or equal to Y
  383. X == Y -> X equals Y
  384. X ~= Y -> X does not equal Y
  385.  
  386.   [!] When used with strings, they are evaluated based on the string's alphabetical ordering
  387.   [!] Type coercion does not work thus a statement like
  388.  
  389.      1| print("1" == 1)
  390.  
  391.      ^^^
  392.      Returns false, and thus you have to use tonumber() or tostring()
  393.  
  394.   [!] When used with objects, they are not equal if:
  395.        1. The types are different
  396.        2. Refer to different objects
  397. ]]
  398.  
  399. if not myVar == 3 then
  400.   print("Not 2")
  401. elseif myBool and myVar == 2 then
  402.   print("Both!")
  403. elseif not myBool or myVar == 2 then
  404.   print("Maybe")
  405. else
  406.   print("?")
  407. end --[[
  408.   Lua also provides conditional operators:
  409. not
  410. and
  411. or
  412.  
  413.   -"not" we simply evaluate -> "not X" expresses to "true" if X is "false" and vice versa
  414.   -"and" we simply require both sides of the expression to evaluate to true before the statement is fully evaluated as "true"
  415.   -"or" we simply require only one side of the expression to evaluate to true before the statement is fully evaluated as "true"
  416.  
  417.   [!] Note when we ever "not" a Number or String, Lua always returns "false"
  418. ]]
  419.  
  420. if myVar and (not true or myVar == 3) then
  421.  
  422. end --[[
  423.   Take this hypothetical piece of conditional statement.
  424.  
  425.   Since operations/operators have precedence, we can use "()" to group expression to make them
  426.   be evaluated first, thus increasing their precedence artificially.
  427.  
  428.   This format also increases readability
  429. ]]
  430.  
  431. myLuaElvis = 1+1==2 and "Yes" or "No" --[[
  432.   Implementation of a basic Ternary operation in Lua.
  433.  
  434.   Lua does not support Ternary or Elvis Operators for one line IF-ELSE
  435. ]]

  438. --[[LOOPS]]--
  439.  
  440. while myBool do print("Hello World") end --[[
  441.   This is a basic LOOP. A while loop uses a boolean condition to make the loop run.
  442.  
  443.   Only until the expression evaluates to false, does the loop exit.
  444.  
  445.   They follow this format:
  446.  
  447. 1| while [bool-expr] do <- Loop declaration
  448. 2|   todo <- Blocks of code to execute
  449. 3| end <- Signifies the end of the loop definition
  450.  
  451. ^^^^
  452. for [bool-expr], you can treat it like any expression you would use in IF-ELSEIF-ELSE statements.
  453. ]]
  454.  
  455. repeat print("Hello world") until myBool --[[
  456.   Similar to a while-loop except that it runs until the condition after "until" becomes true
  457.  
  458.   Follows this format:
  459.  
  460. 1| repeat <- Loop declaration
  461. 2|   todo <- Blocks of code to execute
  462. 3| until [bool-expr] <- Signified the end condition
  463.  
  464.   [!] This loop is similar to a DO-WHILE loop in other languages in which it is guaranteed to run at least once
  465.    and then check the condition after the first iteration.
  466. ]]
  467.  
  468. for i = 1, 10, 1 do print(i) end --[[
  469.   This is a numerical-step for-loop, it follows a similar fashion in many other languages.
  470.  
  471.   It uses two formats:
  472.  
  473.      1| for var = start, end, step do
  474.      2|  todo <- Code to execute
  475.      3| end <- Loop End Declaration
  476.  
  477.      ^^^
  478.      Start -> the start counter
  479.      End   -> the end value to reach (if "var" is greater than end, the loop exits)
  480.      Step  -> How much to increment per iteration. This can also be negative, but will mean that the loop
  481.              will end only if "var" is less than "End"
  482.  
  483.      1| for var = start, end do
  484.      2|   todo <- Code to execute
  485.      3| end <- Loop End Declaration
  486.  
  487.      ^^^
  488.      This modification assumes "step" to be "+1"
  489.  
  490.  
  491. In some languages like Java, C/C++, JavaScript, this kind of loop can be represented as:
  492.  
  493. ----- Lua -----   |  ----- Other Langs. -----
  494.  
  495. 1| for i=0,10 do  |  1| for(var i=0;i<=10;i+=1){
  496. 2| end         |  2| }
  497.  
  498. [!] Scoping does not allow the variable in a numerical step loop to be used outside of the loop block
  499. ]]
  500.  
  501. myTable = {1,2,3,4}
  502. for key,val in ipairs(myTable) do print(key.." "..val) end --[[
  503.   This is an iterator based loop. Primarily used for looping through a data structure like that of
  504.   a Table/Array.
  505.  
  506.   They follow this format:
  507.  
  508. 1| for var1,var2,var3...varN in [itr-expr] do
  509. 2|  todo <- Code to execute
  510. 3| end <- Loop End Declaration
  511.  
  512.   [itr-expr] represents a valid iterator function. Most commonly we use the "ipairs()"
  513. function to retrieve a key and value pair.
  514. ]]
  515.  
  516. --[[
  517.   Control statements for loops are:
  518.  
  519. break -> jump out of the innermost loop
  520. goto [expr] -> tell the program to go to this point in the program
  521.    - Goto a expression can be marked with "::[expr-name]::"
  522.  
  523.    Like so:
  524.  
  525.    1| while true do
  526.    2|  if someVar > 100 then
  527.    3| goto outside
  528.    4|  end
  529.    5| end
  530.    6| ::outside::
  531.  
  532. ]]
  533. myVarA = 1
  534. while true do
  535.   myVarA = myVarA + 1
  536.   if myVarA > 5 then break end
  537. end -- BREAK PATTERN DEMONSTRATION
  538.  
  539. ---------------------------
  540. --[[ SECTION FUNCTIONS ]]--
  541. ---------------------------
  542. function fun()
  543.   print("Function!")
  544. end --[[
  545.   Functions declare snippets of code that can be called and reused later.
  546.  
  547.   Functions follow the format of:
  548.  
  549.    1| function functionName(args) <- The function start
  550.    2|  todo <- Stuffs to execute
  551.    3| end <- The end of the function declaration (similar to a conditional block) ends with "end"
  552.  
  553.    ^^^^^
  554.    The function start also contains "args" which can be left blank or provided with a
  555.    required argument to take. This is called function input/argument.
  556.  
  557.   Within the stuffs to execute section, you can write code like any other part of the program we talked
  558.   about before. However, if you use parameters (described below), those are treated as a variable
  559. ]]
  560.  
  561. function add(a, b)
  562.   return a + b
  563. end
  564. print(add(1, 2)) --[[
  565.   Functions can take input via adding variable names in between the parentheses.
  566.  
  567.   This now means add() requires two arguments in order to work
  568.  
  569.   The "return" statement is straightforward, it represents a possible result
  570.   the function wants to return. However we cannot specify anything else after
  571.   this "return" statement
  572.    -If no "return" statement is found, the function will not return a resultant:
  573.        function m(a,b) c = a+b end
  574.        print(m(1,2)) -> Nothing or ""
  575.    -This is similar in many other languages: Kotlin/Java, C/C++, JavaScript/TypeScript
  576.    -The resultant should be treated as a "raw" value
  577.    -The only exception to no other statements after "return" is for conditional blocks:
  578. ]]
  579. function conditionalReturns(a,b)
  580.   if a + b > 10 then return a + b - 10
  581.   elseif  a + b == 10 then return a + b + 10
  582.   end
  583.   return a+b
  584. end --[[
  585.   Conditional returns can have their return statements wrapped up.
  586.  
  587.   Note the ending "return a+b". This is the "else" case. You can write
  588.   an "else" case within the conditional block, but it is not necessary,
  589.   since if all other cases fail, the program will jump out and use the
  590.   last return.
  591. ]]
  592.  
  593. fun() --[[
  594.   Similar to calling print() and type()
  595.   calling any other function is as simple as:
  596.  
  597.    functionName(args?)
  598.  
  599.    args? -> If argument required
  600. ]]
  601.  
  602. myFunction = function(a,b) return a * b end
  603. print(myFunction(a,b)) -- We can also declare functions to variables and call them like this
  604.  
  605. --[[
  606.   Builtin functions are functions that come pre bundled with Lua.
  607.   Previously the "type()" and "print()" were builtin functions.
  608.   But there are more.
  609.  
  610.   "tonumber","tointeger" etc are also built in
  611.  
  612.   You can find them by starting with "math.", "string."
  613.  
  614.   There are a lot, so we will not mention all of them.
  615.   Check them out here:
  616.   http://lua-users.org/wiki/TutorialDirectory
  617.  
  618.   Examples:
  619. ]]
  620. string.len("Hello World") -- Gets the length of a provided string
  621. string.upper("yooooo") -- Makes all characters in a string to uppercase (abc -> ABC)
  622. string.lower("yoink")  -- Makes all characters in a string to lowercase (ABC -> abc)
  623. math.min(2, 30000)     -- Returns the smallest of the two
  624. math.max(2, 30000)     -- Returns the largest of the two
  625. math.random(0, 100)    -- Returns a pseudorandom number from [start, end) where "end" is excluded and "start" is included
  626.

  629. --[[ SECTION METAPROGRAMMING & METAMETHODS ]]--
  630. -----------------------------------------------
  631. --[[
  632.   Metamethods in Lua enables the programmer to overload certain functionalities in Lua objects.
  633.    -Similar to the concept of operator overloading in a language like C++
  634.  
  635.   Metables is a regular Table data structure that holds metamethods that can be called when specific
  636.   events in Lua are called. Like for example addition ("+" -> __add)
  637.  
  638.   To make a metatable work, we can use the following exemplar:
  639. ]]
  640. local myObj = { attribute = ":)" }
  641. local myMetaTable = {
  642.   __add = function(lhs, rhs) -- here lhs -> LeftHandSide & rhs -> RightHandSide
  643.    return { value = lhs.attribute .. rhs.attribute } -- the desired operation for when the "+" or addition operation is performed on
  644.                                                      -- two objects of myObj
  645.   end
  646. }
  647. setmetatable(myObj, myMetaTable) -- tell Lua we have a new metamethod we want to add
  648. --[[
  649.   Thus we can now do things like
  650.  
  651.    1| local myObjFinal = myObj + myObj
  652.    2| print(myObjFinal)
  653.  
  654. ]]
  655.  
  656. --[[
  657.   You can find a list of most overloadable metamethods that you can use here:
  658.   http://www.lua.org/manual/5.4/manual.html#2.4
  659. ]]
  660.

  663. --[[ SECTION MODULES ]]--
  664. -------------------------
  665. --[[
  666.   [!] THIS IS TARGETED FOR 5.2 AND ABOVE
  667.  
  668.   Modules are the most basic way for Lua code to be bundled and used together.
  669.  
  670.   You declare in the following fashion:
  671.  
  672.  -- BEGIN: file testModule.lua --
  673.  
  674.   1| local testModule = {} -- Note you don't specifically have to name the module the same as the file name
  675.   2| function testModule.someFun() print("This is a test") end
  676.   3| return testModule
  677.  
  678. -- END: file testModule.lua
  679.   To use this module in some other file, we use
  680.   the "require" keyword:
  681. -- BEGIN: file testMyModule.lua --
  682.  
  683.   1| tM = require 'testModule'
  684.   2| tM.someFun() -- prints "This is a test"
  685.  
  686. Modules are cached in the global table "package.loaded".
  687. -this also provides us the ability to reload:
  688.  
  689.    1| package.loaded.someModule = nil
  690.    2| m = require 'someModule'
  691.  
  692. You can think of this as a JavaScript Module.exports:
  693.  
  694. 1| const pkgCode = require("myCodeHere")
  695.  
  696. Or you can think of it as a C++ Namespace, Java Import/Class/Module
  697.  
  698. You can also selectively expose modules contents:
  699.  
  700. 1) Declare certain functions as local
  701. 2) This format:
  702.  
  703.    1| local function f1() ... end
  704.    2| local function f2() ... end
  705.    3| return { f1 = f1, f2 = f2 }
  706.  
  707. 3) Combine 1 & 2:
  708.  
  709.    1| local me = {}
  710.    2| local function f1() ... end
  711.    3| local function f2() ... end
  712.    4| me.f1 = f1
  713.    5| me.f2 = f2
  714.    6| return me
  715.  
  716. There are some other ways: http://lua-users.org/wiki/ModulesTutorial (using Environment Variables)
  717. ]]
  718.

  721. --[[ SECTION OBJECT ORIENTATION ]]--
  722. ------------------------------------
  723. --[[
  724.   Even though Lua is not OOP and there are no Class constructs, we can imitate using metatables.
  725.  
  726.   We can achieve this by creating a table that represents a "class" and have metatable attributes like
  727.   "__index" and then we can use modular functions to create instance methods
  728.  
  729.   We often use the "self" semi-keyword to refer to the current instance.
  730.  
  731.   When we create a function with the first argument as "self" Lua automatically has a feature where if you want
  732.   access a member method within the instance, you can use the ":" operator to work on this instance without
  733.   having to call:
  734.  
  735.    1| myInstance.func(myInstance, args)
  736.  
  737.   Instead you can just do:
  738.  
  739.    2| myInstance:func(args)
  740.  
  741.   For example:
  742. ]]
  743.  
  744. local ExampleClass = {}
  745. ExampleClass.__index = ExampleClass -- if default lookup fails we daful to table lookups for method names
  746. function ExampleClass.new(init) -- a function equivalent to a constructor
  747.   local self = setmetatable({}, ExampleClass)
  748.   self.value = init
  749.   return self
  750. end
  751. function ExampleClass.set_val(self, val)
  752.   self.value = val
  753. end
  754. --Usage--
  755. local exampleClass_E = ExampleClass.new(5)
  756. exampleClass_E:set_val(1) -- value is now 1
  757. --[[
  758.   This seems quite similar to how some other languages do their OOP-ish
  759.   modularity, like JavaScript, Python, etc..
  760. ]]
